20  package org.apache.hadoop.hbase.master;
21  
22  import org.apache.hadoop.conf.Configurable;
23  import org.apache.hadoop.fs.Path;
24  
25  /**
26   * Interface for the log cleaning function inside the master. By default, three
27   * cleaners <code>TimeToLiveLogCleaner</code>,  <code>ReplicationLogCleaner</code>,
28   * <code>SnapshotLogCleaner</code> are called in order. So if other effects are
29   * needed, implement your own LogCleanerDelegate and add it to the configuration
30   * "hbase.master.logcleaner.plugins", which is a comma-separated list of fully
31   * qualified class names. LogsCleaner will add it to the chain.
32   *
33   * HBase ships with LogsCleaner as the default implementation.
34   *
35   * This interface extends Configurable, so setConf needs to be called once
36   * before using the cleaner.
37   * Since LogCleanerDelegates are created in LogsCleaner by reflection. Classes
38   * that implements this interface should provide a default constructor.
39   */
40  public interface LogCleanerDelegate extends Configurable {
41  
42    /**
43     * Should the master delete the log or keep it?
44     * @param filePath full path to log.
45     * @return true if the log is deletable, false if not
46     */
47    public boolean isLogDeletable(Path filePath);
48  }
